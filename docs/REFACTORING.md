          <Badge onClick={handleClearSearch}>
            Filtered: {booksToDisplay.length} result...

      {/* UI BOOKS DISPLAY (10 lignes) */}
      {booksToDisplay.length > 0 ? (
        <div className="grid grid-cols-1 sm:grid-cols-2 ...">

          ))}
        </div>

              : 'No books available on this page.'

              Clear search filter

          )}
